' Adds date in next blank cell in column A 
' The column can be modified to suit your needs 

Sub AddDateToAllSheets()
    Dim ws As Worksheet
    Dim lastRow As Long
    
    For Each ws In ThisWorkbook.Worksheets
        lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
        
        ' Add "Next_month" to the next empty cell in column A
        ws.Cells(lastRow + 1, 1).Value = "May'26"
    Next ws
End Sub
